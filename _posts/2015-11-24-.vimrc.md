Here is my .vimrc

~~~
set number
set ruler
set laststatus=2
set enc=utf-8
set history=1000

syntax on
filetype on

set autoindent
set smartindent
set cindent
set tabstop=4
set shiftwidth=4
set softtabstop=4
set expandtab
set smarttab

set showmatch
set nohls
set hlsearch
set incsearch

au FileType ruby setl shiftwidth=2 softtabstop=2 et
au FileType python setl shiftwidth=2 softtabstop=2 et
~~~
